\def\explainmathbox{%
\rule{8pt}{0pt}\parbox{4.5cm}{
\footnotesize\smallskip
Formula di Johnson ($\lambda<\lambda_\mathrm{t}$)
\[\sigma_\mathrm{cr}=
  \sigma_\mathrm{s}\left(1-\frac{\lambda^2}
{4\pi^2E}\sigma_\mathrm{s}\right) 
\] 
Formula di Eulero ($\lambda\geq\lambda_\mathrm{t}$)
\[\sigma_\mathrm{cr}=
  \dfrac{\pi^2 E}{\lambda^2} 
\] 
Snellezza di transizione:
\[\lambda_\mathrm{t}=
  \pi\sqrt{\dfrac{2E}{\sigma_\mathrm{s}}} 
\]
}
}
